\documentclass[border=0.3cm]{standalone}

\usepackage[scaled]{helvet}
\renewcommand\familydefault{\sfdefault}
\usepackage[T1]{fontenc}

\usepackage{tikz}
\usetikzlibrary{arrows.meta, calc, positioning}

\definecolor{red1}{HTML}{F9665E}
\definecolor{gold1}{HTML}{F1A226}
\definecolor{green1}{HTML}{6DDB5C}
\definecolor{purp1}{HTML}{A44694}

\begin{document}
\begin{tikzpicture}

% ============================================================
% CENTRAL CONSTRUCT LIBRARY (stacked)
% ============================================================
\def\ntscale{0.03}  % cm per nt
\pgfmathsetmacro{\fw}{20*\ntscale}     % flank width
\pgfmathsetmacro{\bkw}{30*\ntscale}    % barcode width
\pgfmathsetmacro{\tpd}{130*\ntscale}   % total pad+design width
\def\rh{0.25}  % row height
\def\rg{0.06}  % row gap
\def\nrows{4}

% Total width (constant across rows)
\pgfmathsetmacro{\totalw}{\fw+\tpd+\bkw+\fw}
\pgfmathsetmacro{\startx}{-\totalw/2}

% Draw 4 constructs with varying design/padding splits
% design widths in cm (ascending)
\foreach \dcm/\i in {1.8/0, 0.6/1, 2.4/2, 1.2/3} {
    \pgfmathsetmacro{\yy}{(\nrows-1)*(\rh+\rg)/2 - \i*(\rh+\rg)}
    \pgfmathsetmacro{\pcm}{\tpd - \dcm}
    \pgfmathsetmacro{\x}{\startx}
    % 5' flank
    \draw[thick, fill=red1!30] (\x, {\yy+\rh/2}) rectangle ({\x+\fw}, {\yy-\rh/2});
    \pgfmathsetmacro{\x}{\x+\fw}
    % padding
    \draw[thick, fill=green1!35] (\x, {\yy+\rh/2}) rectangle ({\x+\pcm}, {\yy-\rh/2});
    \pgfmathsetmacro{\x}{\x+\pcm}
    % design
    \draw[thick, fill=purp1!35] (\x, {\yy+\rh/2}) rectangle ({\x+\dcm}, {\yy-\rh/2});
    \pgfmathsetmacro{\x}{\x+\dcm}
    % barcode
    \draw[thick, fill=gold1!40] (\x, {\yy+\rh/2}) rectangle ({\x+\bkw}, {\yy-\rh/2});
    \pgfmathsetmacro{\x}{\x+\bkw}
    % 3' flank
    \draw[thick, fill=red1!30] (\x, {\yy+\rh/2}) rectangle ({\x+\fw}, {\yy-\rh/2});
}

% Compute bounding box midpoints for arrow targets (use middle row average)
% Use the x-ranges across the stack for each segment type
\pgfmathsetmacro{\xa}{\startx}
\pgfmathsetmacro{\xFlankEnd}{\xa+\fw}
% Padding region spans from xFlankEnd to xFlankEnd+pad (varies per row, use average)
\pgfmathsetmacro{\midPadAvg}{\xFlankEnd + \tpd*0.25}  % ~middle of padding region
% Design region center (varies, use average)
\pgfmathsetmacro{\midDesignAvg}{\xFlankEnd + \tpd*0.75}
% Barcode
\pgfmathsetmacro{\xBCstart}{\xFlankEnd + \tpd}
\pgfmathsetmacro{\midBC}{(\xBCstart + \xBCstart + \bkw)/2}
% Right flank
\pgfmathsetmacro{\xRFstart}{\xBCstart + \bkw}
\pgfmathsetmacro{\midFlankR}{(\xRFstart + \xRFstart + \fw)/2}
% Stack vertical extent
\pgfmathsetmacro{\stackTop}{(\nrows-1)*(\rh+\rg)/2 + \rh/2}
\pgfmathsetmacro{\stackBot}{-(\nrows-1)*(\rh+\rg)/2 - \rh/2}

% ============================================================
% NODE POSITIONS
% ============================================================
\def\ny{2.0}   % vertical center for top nodes
\def\by{-2.0}  % vertical center for bottom nodes
\def\nlx{-2.8} % horizontal center for left nodes
\def\nrx{2.8}  % horizontal center for right nodes
\def\nw{2.8}   % node box width

% Arm intermediate y-positions
\def\armTop{0.9}     % horizontal segment y for top nodes
\def\armBot{-0.9}    % horizontal segment y for bottom nodes

% ============================================================
% 1. DESIGN (top-left)
% ============================================================
\begin{scope}[shift={(\nlx, \ny)}]
    \node[font=\bfseries, anchor=south] at (0, 0.9) {1. Design};

    \node[font=\small, anchor=center] at (0, 0.35)
        {\texttt{..GRY..} $\to$ \texttt{..GAC..}};
    \node[font=\scriptsize\itshape, anchor=center] at (0, -0.15)
        {Resolve ambiguous bases};

    \draw[thick, rounded corners=3pt] (-\nw/2, -0.5) rectangle (\nw/2, 1.15);
\end{scope}

% 3-stage arrow: down from node, right to segment x, down to construct
\draw[thick, -stealth]
    (\nlx, {\ny-0.5}) -- (\nlx, \armTop)
    -- (\midDesignAvg, \armTop)
    -- (\midDesignAvg, \stackTop);

% ============================================================
% 2. PADDING (bottom-left)
% ============================================================
\begin{scope}[shift={(\nlx, \by)}]
    \node[font=\bfseries, anchor=north] at (0, -1.1) {2. Padding};

    % Paired-bar schematic
    \def\bh{0.18}
    \def\bg{0.08}
    \foreach \pl/\dl/\i in {1.1/0.35/0, 0.75/0.7/1, 0.35/1.1/2} {
        \pgfmathsetmacro{\yy}{0.65 - \i*(\bh+\bg)}
        \draw[thick, fill=green1!35] (-1.3, \yy) rectangle ({-1.3+\pl}, {\yy-\bh});
        \draw[thick, fill=purp1!35] ({1.3-\dl}, \yy) rectangle (1.3, {\yy-\bh});
        \draw[thin, purp1!50] ({-1.3+\pl}, {\yy-\bh/2}) -- ({1.3-\dl}, {\yy-\bh/2});
    }

    \node[font=\scriptsize\itshape, anchor=center] at (0, -0.2)
        {Inverse-rank pairing};
    \node[font=\scriptsize\itshape, anchor=center] at (0, -0.55)
        {GC/GU constraints};

    \draw[thick, rounded corners=3pt] (-\nw/2, -1.35) rectangle (\nw/2, 0.9);
\end{scope}

% 3-stage arrow: up from node, right to segment x, up to construct
\draw[thick, -stealth]
    (\nlx, {\by+0.9}) -- (\nlx, \armBot)
    -- (\midPadAvg, \armBot)
    -- (\midPadAvg, \stackBot);

% ============================================================
% 3. BARCODE (bottom-right)
% ============================================================
\begin{scope}[shift={(\nrx, \by)}]
    \node[font=\bfseries, anchor=north] at (0, -1.1) {3. Barcode};

    \def\bh{0.18}
    \def\bg{0.08}
    \foreach \dl/\bl/\i in {0.35/1.1/0, 0.7/0.75/1, 1.1/0.35/2} {
        \pgfmathsetmacro{\yy}{0.65 - \i*(\bh+\bg)}
        \draw[thick, fill=purp1!35] (-1.3, \yy) rectangle ({-1.3+\dl}, {\yy-\bh});
        \draw[thick, fill=gold1!40] ({1.3-\bl}, \yy) rectangle (1.3, {\yy-\bh});
        \draw[thin, purp1!50] ({-1.3+\dl}, {\yy-\bh/2}) -- ({1.3-\bl}, {\yy-\bh/2});
    }

    \node[font=\scriptsize\itshape, anchor=center] at (0, -0.2)
        {Inverse-rank pairing};
    \node[font=\scriptsize\itshape, anchor=center] at (0, -0.55)
        {Hamming $\geq 2$, GC/GU constr.};

    \draw[thick, rounded corners=3pt] (-\nw/2, -1.35) rectangle (\nw/2, 0.9);
\end{scope}

% 3-stage arrow: up from node, left to segment x, up to construct
\draw[thick, -stealth]
    (\nrx, {\by+0.9}) -- (\nrx, \armBot)
    -- (\midBC, \armBot)
    -- (\midBC, \stackBot);

% ============================================================
% 4. FLANKING (top-right)
% ============================================================
\begin{scope}[shift={(\nrx, \ny)}]
    \node[font=\bfseries, anchor=south] at (0, 0.45) {4. Flanking};

    \node[font=\small, anchor=center] at (0, 0.05)
        {\textit{T7 promoter $+$ primers}};

    \draw[thick, rounded corners=3pt] (-\nw/2, -0.3) rectangle (\nw/2, 0.7);
\end{scope}

% 3-stage arrow: down from node, left to segment x, down to construct
\draw[thick, -stealth]
    (\nrx, {\ny-0.3}) -- (\nrx, \armTop)
    -- (\midFlankR, \armTop)
    -- (\midFlankR, \stackTop);

\end{tikzpicture}
\end{document}
