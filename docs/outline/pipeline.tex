% ============================================================
% CENTRAL CONSTRUCT LIBRARY (stacked)
% ============================================================
\def\ntscale{0.03}  % cm per nt
\pgfmathsetmacro{\fw}{20*\ntscale}     % flank width
\pgfmathsetmacro{\bkw}{30*\ntscale}    % barcode width
\pgfmathsetmacro{\tpd}{130*\ntscale}   % total pad+design width
\def\rh{0.25}  % row height
\def\rg{0.06}  % row gap
\def\nrows{4}

% Total width (constant across rows)
\pgfmathsetmacro{\totalw}{\fw+\tpd+\bkw+\fw}
\pgfmathsetmacro{\startx}{-\totalw/2}

% Draw 4 constructs with varying design/padding splits
% design widths in cm (ascending)
\foreach \dcm/\i in {1.8/0, 0.6/1, 2.4/2, 1.2/3} {
    \pgfmathsetmacro{\yy}{(\nrows-1)*(\rh+\rg)/2 - \i*(\rh+\rg)}
    \pgfmathsetmacro{\pcm}{\tpd - \dcm}
    \pgfmathsetmacro{\x}{\startx}
    % 5' flank
    \draw[thick, fill=red1!30] (\x, {\yy+\rh/2}) rectangle ({\x+\fw}, {\yy-\rh/2});
    \pgfmathsetmacro{\x}{\x+\fw}
    % padding
    \draw[thick, fill=green1!35] (\x, {\yy+\rh/2}) rectangle ({\x+\pcm}, {\yy-\rh/2});
    \pgfmathsetmacro{\x}{\x+\pcm}
    % design
    \draw[thick, fill=purp1!35] (\x, {\yy+\rh/2}) rectangle ({\x+\dcm}, {\yy-\rh/2});
    \pgfmathsetmacro{\x}{\x+\dcm}
    % barcode

    \draw[thick, fill=red1!30] (\x, {\yy+\rh/2}) rectangle ({\x+\fw}, {\yy-\rh/2});
}

% Compute bounding box midpoints for arrow targets (use middle row average)

\pgfmathsetmacro{\midPadAvg}{\xFlankEnd + \tpd*0.25}  % ~middle of padding region
% Design region center (varies, use average)
\pgfmathsetmacro{\midDesignAvg}{\xFlankEnd + \tpd*0.75}

% 2. PADDING (bottom-left)
% ============================================================
\begin{scope}[shift={(\nlx, \by)}]
    \node[font=\bfseries, anchor=north] at (0, -1.55) {2. Padding};

    % Two length bins: all rows same total width, gap at pad|design boundary
    \def\bh{0.16}
    \def\bg{0.05}
    \def\binsep{0.14}
    \def\tw{1.6}       % total width (pad+gap+design)

    \def\lx{-0.8}      % left edge (aligned for all rows)

    % Bin 1: 3 rows, pad=0.35, design=1.0
    \def\pdA{0.35}
    \pgfmathsetmacro{\dwA}{\tw - \gapmid - \pdA}
    \foreach \i in {0, 1, 2} {
        \pgfmathsetmacro{\yy}{0.7 - \i*(\bh+\bg)}
        \draw[thick, fill=green1!40] (\lx, \yy) rectangle ({\lx+\pdA}, {\yy-\bh});
        \draw[thick, fill=purp1!40] ({\lx+\pdA+\gapmid}, \yy) rectangle ({\lx+\tw}, {\yy-\bh});
        \draw[thin] ({\lx+\pdA}, {\yy-\bh/2}) -- ({\lx+\pdA+\gapmid}, {\yy-\bh/2});
    }

    % Bin 2: 2 rows, pad=0.75, design=0.6 (same total)
    \def\pdB{0.75}
    \pgfmathsetmacro{\dwB}{\tw - \gapmid - \pdB}
    \pgfmathsetmacro{\bintwotop}{0.7 - 3*(\bh+\bg) - \binsep}
    \foreach \i in {0, 1} {
        \pgfmathsetmacro{\yy}{\bintwotop - \i*(\bh+\bg)}
        \draw[thick, fill=green1!40] (\lx, \yy) rectangle ({\lx+\pdB}, {\yy-\bh});
        \draw[thick, fill=purp1!40] ({\lx+\pdB+\gapmid}, \yy) rectangle ({\lx+\tw}, {\yy-\bh});
        \draw[thin] ({\lx+\pdB}, {\yy-\bh/2}) -- ({\lx+\pdB+\gapmid}, {\yy-\bh/2});
    }

    % Sort arrows: single grey arrow spanning all rows on each side
    \pgfmathsetmacro{\arrTop}{0.7 - \bh/2}
    \pgfmathsetmacro{\arrBot}{\bintwotop - 1*(\bh+\bg) - \bh/2}
    \pgfmathsetmacro{\arrL}{\lx - 0.2}
    \pgfmathsetmacro{\arrR}{\lx + \tw + 0.2}
    \draw[gray, thin, -stealth] (\arrL, \arrTop) -- (\arrL, \arrBot);
    \draw[gray, thin, stealth-] (\arrR, \arrTop) -- (\arrR, \arrBot);

    % Arrow labels
    \node[font={\fontsize{5.5}{6.5}\selectfont}, gray, rotate=90, anchor=center] at ({\arrL - 0.15}, {(\arrTop+\arrBot)/2})

    \node[font={\fontsize{5.5}{6.5}\selectfont}, gray, rotate=-90, anchor=center] at ({\arrR + 0.15}, {(\arrTop+\arrBot)/2})
        {Reads};



\end{scope}

    \node[font=\bfseries, anchor=north] at (0, -1.1) {3. Barcode};

    \def\bh{0.16}
    \def\bg{0.05}
    \def\tw{1.6}       % total width (pad+design+gap+barcode)
    \def\gapmid{0.25}  % gap between joined construct and barcode
    \def\lx{-0.8}      % left edge
    \def\pdtotal{1.1}  % pad+design total width (constant)

        % connecting line

% ============================================================
% HAIRPIN SCHEMATIC (between nodes 2 and 3)
% ============================================================
% ChimeraX NDB colors: A=red, C=yellow, G=green, U=cyan
\definecolor{baseA}{HTML}{E8655A}
\colorlet{baseC}{gold1}
\colorlet{baseG}{green1}
\definecolor{baseU}{HTML}{89CFF0}

\begin{scope}[shift={(0, \by)}]
    % Hairpin: 4bp stem + 4nt loop
    \def\nr{0.09}       % nucleotide circle radius
    \def\stemgap{0.40}  % horizontal gap between stem strands
    \def\stepy{0.26}    % vertical step between base pairs

    % Stem base pairs (bottom to top): G-C, C-G, A-U, G-C
    % Left strand (5' to 3', bottom to top)
    \foreach \col/\lbl/\i in {baseG/G/0, baseC/C/1, baseA/A/2, baseG/G/3} {
        \pgfmathsetmacro{\yy}{-0.3 + \i*\stepy}
        % left base
        \draw[thick, fill=\col!60] ({-\stemgap/2}, \yy) circle (\nr);
        % right base (complement)
    }
    % Right strand (3' to 5', bottom to top): C, G, U, C
    \foreach \col/\lbl/\i in {baseC/C/0, baseG/G/1, baseU/U/2, baseC/C/3} {
        \pgfmathsetmacro{\yy}{-0.3 + \i*\stepy}
        \draw[thick, fill=\col!60] ({\stemgap/2}, \yy) circle (\nr);
        % base pair line
        \draw[thin] ({-\stemgap/2+\nr}, \yy) -- ({\stemgap/2-\nr}, \yy);
    }
    % Backbone lines (stem)
    \foreach \i in {0, 1, 2} {
        \pgfmathsetmacro{\ya}{-0.3 + \i*\stepy}
        \pgfmathsetmacro{\yb}{-0.3 + (\i+1)*\stepy}
        \draw[thick] ({-\stemgap/2}, {\ya+\nr}) -- ({-\stemgap/2}, {\yb-\nr});
        \draw[thick] ({\stemgap/2}, {\ya+\nr}) -- ({\stemgap/2}, {\yb-\nr});
    }

    % Loop: 6 points on a circular arc, equally spaced by angle (equal chords).
    % Positions 0,5 = stem tops; 1-4 = loop bases.
    % Symmetric about 90°. Step = 30°, arc = 150°.
    % Stem-L at 165°, stem-R at 15°. r = stemgap / (2 sin(75°)).
    \pgfmathsetmacro{\loopbase}{-0.3 + 3*\stepy}
    \pgfmathsetmacro{\ltopY}{\loopbase + \nr}
    \def\loopstep{50}
    \pgfmathsetmacro{\stemAngL}{90 + 5*\loopstep/2}
    \def\loopr{0.305}  % fixed loop radius (decoupled from stem gap)
    \pgfmathsetmacro{\cx}{0}
    \pgfmathsetmacro{\cy}{\ltopY - \loopr*sin(\stemAngL) - 0.06}
    % Backbone lines first (so circles render on top)
    \pgfmathsetmacro{\fAng}{\stemAngL - \loopstep}
    \draw[thick] ({-\stemgap/2}, \ltopY) -- ({\cx + \loopr*cos(\fAng)}, {\cy + \loopr*sin(\fAng)});
    \foreach \idxA/\idxB in {1/2, 2/3, 3/4} {
        \pgfmathsetmacro{\angA}{\stemAngL - \idxA*\loopstep}
        \pgfmathsetmacro{\angB}{\stemAngL - \idxB*\loopstep}
        \draw[thick] ({\cx+\loopr*cos(\angA)}, {\cy+\loopr*sin(\angA)})
                  -- ({\cx+\loopr*cos(\angB)}, {\cy+\loopr*sin(\angB)});
    }
    \pgfmathsetmacro{\lAng}{\stemAngL - 4*\loopstep}
    \draw[thick] ({\cx + \loopr*cos(\lAng)}, {\cy + \loopr*sin(\lAng)}) -- ({\stemgap/2}, \ltopY);
    % Loop base circles (on top of backbone)
    \foreach \col/\idx in {baseU/1, baseU/2, baseC/3, baseG/4} {
        \pgfmathsetmacro{\ang}{\stemAngL - (\idx)*\loopstep}
        \pgfmathsetmacro{\lx}{\cx + \loopr*cos(\ang)}
        \pgfmathsetmacro{\ly}{\cy + \loopr*sin(\ang)}
        \draw[thick, fill=\col!60] (\lx, \ly) circle (\nr);
    }

    % Dangling ends at bottom
    \pgfmathsetmacro{\botY}{-0.3 - \nr}
    \draw[thick] ({-\stemgap/2}, \botY) -- ({-\stemgap/2}, {\botY - 0.15});
    \draw[thick] ({\stemgap/2}, \botY) -- ({\stemgap/2}, {\botY - 0.15});
    \node[font=\tiny, anchor=north] at ({-\stemgap/2}, {\botY - 0.15}) {5'};
    \node[font=\tiny, anchor=north] at ({\stemgap/2}, {\botY - 0.15}) {3'};

    % Caption
    \node[font=\scriptsize\itshape, anchor=north, align=center] at (0, {\botY - 0.4})
        {GC-constrained\\random hairpins};

    % Arrows pointing to padding (left) and barcode (right) nodes
    \draw[thin, -stealth] (-0.5, {(\loopbase + \botY)/2}) -- ({-\stemgap/2 - 0.8}, {(\loopbase + \botY)/2});
    \draw[thin, -stealth] (0.5, {(\loopbase + \botY)/2}) -- ({\stemgap/2 + 0.8}, {(\loopbase + \botY)/2});
\end{scope}
