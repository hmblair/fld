\documentclass[border=0.3cm]{standalone}

\usepackage[scaled]{helvet}
\renewcommand\familydefault{\sfdefault}
\usepackage[T1]{fontenc}

\usepackage{tikz}
\usetikzlibrary{arrows.meta, calc, positioning}

\definecolor{red1}{HTML}{F9665E}
\definecolor{gold1}{HTML}{F1A226}
\definecolor{green1}{HTML}{6DDB5C}
\definecolor{purp1}{HTML}{A44694}

\begin{document}
\begin{tikzpicture}

% ============================================================
% CENTRAL CONSTRUCT LIBRARY (stacked)
% ============================================================
\def\ntscale{0.03}  % cm per nt
\pgfmathsetmacro{\fw}{20*\ntscale}     % flank width
\pgfmathsetmacro{\bkw}{30*\ntscale}    % barcode width
\pgfmathsetmacro{\tpd}{130*\ntscale}   % total pad+design width
\def\rh{0.25}  % row height
\def\rg{0.06}  % row gap
\def\nrows{4}

% Total width (constant across rows)
\pgfmathsetmacro{\totalw}{\fw+\tpd+\bkw+\fw}
\pgfmathsetmacro{\startx}{-\totalw/2}

% Draw 4 constructs with varying design/padding splits
% design widths in cm (ascending)
\foreach \dcm/\i in {1.8/0, 0.6/1, 2.4/2, 1.2/3} {
    \pgfmathsetmacro{\yy}{(\nrows-1)*(\rh+\rg)/2 - \i*(\rh+\rg)}
    \pgfmathsetmacro{\pcm}{\tpd - \dcm}
    \pgfmathsetmacro{\x}{\startx}
    % 5' flank
    \draw[thick, fill=red1!30] (\x, {\yy+\rh/2}) rectangle ({\x+\fw}, {\yy-\rh/2});
    \pgfmathsetmacro{\x}{\x+\fw}
    % padding
    \draw[thick, fill=green1!35] (\x, {\yy+\rh/2}) rectangle ({\x+\pcm}, {\yy-\rh/2});
    \pgfmathsetmacro{\x}{\x+\pcm}
    % design
    \draw[thick, fill=purp1!35] (\x, {\yy+\rh/2}) rectangle ({\x+\dcm}, {\yy-\rh/2});
    \pgfmathsetmacro{\x}{\x+\dcm}
    % barcode
    \draw[thick, fill=gold1!40] (\x, {\yy+\rh/2}) rectangle ({\x+\bkw}, {\yy-\rh/2});
    \pgfmathsetmacro{\x}{\x+\bkw}
    % 3' flank
    \draw[thick, fill=red1!30] (\x, {\yy+\rh/2}) rectangle ({\x+\fw}, {\yy-\rh/2});
}

% Compute bounding box midpoints for arrow targets (use middle row average)
% Use the x-ranges across the stack for each segment type
\pgfmathsetmacro{\xa}{\startx}
\pgfmathsetmacro{\xFlankEnd}{\xa+\fw}
% Padding region spans from xFlankEnd to xFlankEnd+pad (varies per row, use average)
\pgfmathsetmacro{\midPadAvg}{\xFlankEnd + \tpd*0.25}  % ~middle of padding region
% Design region center (varies, use average)
\pgfmathsetmacro{\midDesignAvg}{\xFlankEnd + \tpd*0.75}
% Barcode
\pgfmathsetmacro{\xBCstart}{\xFlankEnd + \tpd}
\pgfmathsetmacro{\midBC}{(\xBCstart + \xBCstart + \bkw)/2}
% Right flank
\pgfmathsetmacro{\xRFstart}{\xBCstart + \bkw}
\pgfmathsetmacro{\midFlankR}{(\xRFstart + \xRFstart + \fw)/2}
% Stack vertical extent
\pgfmathsetmacro{\stackTop}{(\nrows-1)*(\rh+\rg)/2 + \rh/2}
\pgfmathsetmacro{\stackBot}{-(\nrows-1)*(\rh+\rg)/2 - \rh/2}

% ============================================================
% NODE POSITIONS
% ============================================================
\def\ny{2.0}   % vertical center for top nodes
\def\by{-2.0}  % vertical center for bottom nodes
\def\nlx{-2.8} % horizontal center for left nodes
\def\nrx{2.8}  % horizontal center for right nodes
\def\nw{2.8}   % node box width

% Arm intermediate y-positions
\def\armTop{0.9}     % horizontal segment y for top nodes
\def\armBot{-0.9}    % horizontal segment y for bottom nodes

% ============================================================
% 1. DESIGN (top-left)
% ============================================================
\begin{scope}[shift={(\nlx, \ny)}]
    \node[font=\bfseries, anchor=south] at (0, 0.9) {1. Design};

    % 5 input design sequences of varying length
    \def\dbh{0.14}
    \def\dbg{0.04}
    \def\dlx{-0.8}
    \foreach \dw/\i in {0.9/0, 0.5/1, 1.3/2, 0.7/3, 1.1/4} {
        \pgfmathsetmacro{\yy}{0.7 - \i*(\dbh+\dbg)}
        \draw[thick, fill=purp1!40] (\dlx, \yy) rectangle ({\dlx+\dw}, {\yy-\dbh});
    }
\end{scope}

% 3-stage arrow: down from node, right to segment x, down to construct
\pgfmathsetmacro{\designNodeBot}{\ny + 0.7 - 4*(0.14+0.04) - 0.14}
\draw[thick, -stealth]
    (\nlx, \designNodeBot) -- (\nlx, \armTop)
    -- (\midDesignAvg, \armTop)
    -- (\midDesignAvg, \stackTop);

% Label on arrow
\node[font=\scriptsize\itshape, fill=white, inner sep=1pt]
    at (\nlx, {(\designNodeBot+\armTop)/2}) {Resolve ambiguous bases};

% ============================================================
% 2. PADDING (bottom-left)
% ============================================================
\begin{scope}[shift={(\nlx, \by)}]
    \node[font=\bfseries, anchor=north] at (0, -1.55) {2. Padding};

    % Two length bins: all rows same total width, gap at pad|design boundary
    \def\bh{0.16}
    \def\bg{0.05}
    \def\binsep{0.14}
    \def\tw{1.6}       % total width (pad+gap+design)
    \def\gapmid{0.25}  % gap between pad and design
    \def\lx{-0.8}      % left edge (aligned for all rows)

    % Bin 1: 3 rows, pad=0.35, design=1.0
    \def\pdA{0.35}
    \pgfmathsetmacro{\dwA}{\tw - \gapmid - \pdA}
    \foreach \i in {0, 1, 2} {
        \pgfmathsetmacro{\yy}{0.7 - \i*(\bh+\bg)}
        \draw[thick, fill=green1!40] (\lx, \yy) rectangle ({\lx+\pdA}, {\yy-\bh});
        \draw[thick, fill=purp1!40] ({\lx+\pdA+\gapmid}, \yy) rectangle ({\lx+\tw}, {\yy-\bh});
        \draw[thin] ({\lx+\pdA}, {\yy-\bh/2}) -- ({\lx+\pdA+\gapmid}, {\yy-\bh/2});
    }

    % Bin 2: 2 rows, pad=0.75, design=0.6 (same total)
    \def\pdB{0.75}
    \pgfmathsetmacro{\dwB}{\tw - \gapmid - \pdB}
    \pgfmathsetmacro{\bintwotop}{0.7 - 3*(\bh+\bg) - \binsep}
    \foreach \i in {0, 1} {
        \pgfmathsetmacro{\yy}{\bintwotop - \i*(\bh+\bg)}
        \draw[thick, fill=green1!40] (\lx, \yy) rectangle ({\lx+\pdB}, {\yy-\bh});
        \draw[thick, fill=purp1!40] ({\lx+\pdB+\gapmid}, \yy) rectangle ({\lx+\tw}, {\yy-\bh});
        \draw[thin] ({\lx+\pdB}, {\yy-\bh/2}) -- ({\lx+\pdB+\gapmid}, {\yy-\bh/2});
    }

    % Sort arrows: single grey arrow spanning all rows on each side
    \pgfmathsetmacro{\arrTop}{0.7 - \bh/2}
    \pgfmathsetmacro{\arrBot}{\bintwotop - 1*(\bh+\bg) - \bh/2}
    \pgfmathsetmacro{\arrL}{\lx - 0.2}
    \pgfmathsetmacro{\arrR}{\lx + \tw + 0.2}
    \draw[gray, thin, -stealth] (\arrL, \arrTop) -- (\arrL, \arrBot);
    \draw[gray, thin, stealth-] (\arrR, \arrTop) -- (\arrR, \arrBot);

    % Arrow labels
    \node[font={\fontsize{5.5}{6.5}\selectfont}, gray, rotate=90, anchor=center] at ({\arrL - 0.15}, {(\arrTop+\arrBot)/2})
        {Reads};
    \node[font={\fontsize{5.5}{6.5}\selectfont}, gray, rotate=-90, anchor=center] at ({\arrR + 0.15}, {(\arrTop+\arrBot)/2})
        {Reads};

    % Caption below, pointing up to bottom padding bar
    \pgfmathsetmacro{\lastRowMid}{\bintwotop - 1*(\bh+\bg) - \bh/2}
    \node[font=\scriptsize\itshape, anchor=north, align=center] (hplabel) at ({\lx + \pdB/2}, -0.65)
        {GC-constrained\\hairpin trains};
    \draw[thin] (hplabel.north) -- ({\lx + \pdB/2}, {\bintwotop - 1*(\bh+\bg) - \bh});


\end{scope}

% 3-stage arrow: up from node, right to segment x, up to construct
\draw[thick, -stealth]
    (\nlx, {\by+0.9}) -- (\nlx, \armBot)

    -- (\midPadAvg, \armBot)
    -- (\midPadAvg, \stackBot);

% ============================================================
% 3. BARCODE (bottom-right)
% ============================================================
\begin{scope}[shift={(\nrx, \by)}]
    \node[font=\bfseries, anchor=north] at (0, -1.1) {3. Barcode};

    \def\bh{0.16}
    \def\bg{0.05}
    \def\tw{1.6}       % total width (pad+design+gap+barcode)
    \def\gapmid{0.25}  % gap between joined construct and barcode
    \def\lx{-0.8}      % left edge
    \def\pdtotal{1.1}  % pad+design total width (constant)
    \def\bcw{0.25}     % barcode width (constant)

    % 5 rows, varying pad/design split, all same total, single bin
    \foreach \pw/\i in {0.35/0, 0.7/1, 0.55/2, 0.25/3, 0.8/4} {
        \pgfmathsetmacro{\yy}{0.7 - \i*(\bh+\bg)}
        \pgfmathsetmacro{\dw}{\pdtotal - \pw}
        % padding (left part of joined bar)
        \draw[thick, fill=green1!40] (\lx, \yy) rectangle ({\lx+\pw}, {\yy-\bh});
        % design (right part of joined bar)
        \draw[thick, fill=purp1!40] ({\lx+\pw}, \yy) rectangle ({\lx+\pdtotal}, {\yy-\bh});
        % barcode (right-aligned)
        \pgfmathsetmacro{\bcx}{\lx+\pdtotal+\gapmid}
        \draw[thick, fill=gold1!40] (\bcx, \yy) rectangle ({\bcx+\bcw}, {\yy-\bh});
        % connecting line
        \draw[thin] ({\lx+\pdtotal}, {\yy-\bh/2}) -- (\bcx, {\yy-\bh/2});
    }

    % Sort arrows
    \pgfmathsetmacro{\arrTop}{0.7 - \bh/2}
    \pgfmathsetmacro{\arrBot}{0.7 - 4*(\bh+\bg) - \bh/2}
    \pgfmathsetmacro{\arrL}{\lx - 0.2}
    \pgfmathsetmacro{\bcx}{\lx+\pdtotal+\gapmid}
    \pgfmathsetmacro{\arrR}{\bcx + \bcw + 0.2}
    \draw[gray, thin, -stealth] (\arrL, \arrTop) -- (\arrL, \arrBot);
    \draw[gray, thin, stealth-] (\arrR, \arrTop) -- (\arrR, \arrBot);

    % Arrow labels
    \node[font={\fontsize{5.5}{6.5}\selectfont}, gray, rotate=90, anchor=center]
        at ({\arrL - 0.15}, {(\arrTop+\arrBot)/2}) {Reads};
    \node[font={\fontsize{5.5}{6.5}\selectfont}, gray, rotate=-90, anchor=center]
        at ({\arrR + 0.15}, {(\arrTop+\arrBot)/2}) {Reads};

    % Caption
    \node[font=\scriptsize\itshape, anchor=north, align=center] (bclabel) at ({\bcx + \bcw/2}, -0.55)
        {Hamming $\geq 2$};
    \draw[thin] (bclabel.north) -- ({\bcx + \bcw/2}, {\arrBot - \bh/2 + 0.02});


\end{scope}

% 3-stage arrow: up from node, left to segment x, up to construct
\draw[thick, -stealth]
    (\nrx, {\by+0.9}) -- (\nrx, \armBot)
    -- (\midBC, \armBot)
    -- (\midBC, \stackBot);

% ============================================================
% 4. FLANKING (top-right)
% ============================================================
\begin{scope}[shift={(\nrx, \ny)}]
    \node[font=\bfseries, anchor=south] at (0, 0.65) {4. Flanking};

    % 5' flank bar, dashed middle, 3' flank bar
    \def\flx{-0.8}     % left edge
    \def\flw{0.3}      % flank bar width
    \def\flh{0.25}     % flank bar height
    \def\dashw{1.0}    % dashed gap width

    % 5' flank
    \draw[thick, fill=red1!30] (\flx, 0) rectangle ({\flx+\flw}, {-\flh});
    \node[font=\tiny] at ({\flx+\flw/2}, {-\flh/2}) {5'};

    % dashed line
    \draw[thick, dash pattern=on 2pt off 2pt] ({\flx+\flw}, {-\flh/2}) -- ({\flx+\flw+\dashw}, {-\flh/2});

    % 3' flank
    \pgfmathsetmacro{\rfx}{\flx+\flw+\dashw}
    \draw[thick, fill=red1!30] (\rfx, 0) rectangle ({\rfx+\flw}, {-\flh});
    \node[font=\tiny] at ({\rfx+\flw/2}, {-\flh/2}) {3'};

    % Label below
    \node[font=\scriptsize\itshape, anchor=north] at ({\flx+\flw+\dashw/2}, {-\flh-0.1})
        {T7 promoter $+$ primers};
\end{scope}

% 3-stage arrow: down from node, left to segment x, down to construct
\pgfmathsetmacro{\flankNodeBot}{\ny - 0.25 - 0.1}
\draw[thick, -stealth]
    (\nrx, \flankNodeBot) -- (\nrx, \armTop)
    -- (\midFlankR, \armTop)
    -- (\midFlankR, \stackTop);

\end{tikzpicture}
\end{document}
